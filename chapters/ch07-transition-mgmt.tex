\section{چالش‌های «روز بعد»}

\begin{enghelabbox}[title=خطرات فوری]
\begin{itemize}
    \item \textbf{خلأ قدرت:} فقدان نهادهای تصمیم‌گیری
    \item \textbf{امنیت:} خطر شورش، غارت، انتقام‌جویی
    \item \textbf{اقتصاد:} بحران ارز، کمبود کالا
    \item \textbf{انتقام:} خطر کشتار عوامل نظام
\end{itemize}
\end{enghelabbox}

\section{نقش نیروهای مسلح}

\begin{itemize}
    \item \textbf{ارتش:} حفظ با پاکسازی فرماندهان + نظارت
    \item \textbf{سپاه:} انحلال یا ادغام تدریجی
    \item \textbf{پلیس:} بازسازی با اولویت
\end{itemize}

\section{عدالت انتقالی}

\begin{olgoobox}[title=اصول عدالت انتقالی]
\begin{enumerate}
    \item \textbf{حقیقت‌یابی:} کمیسیون حقیقت
    \item \textbf{محاکمه:} رهبران اصلی جنایات
    \item \textbf{جبران:} غرامت به قربانیان
    \item \textbf{اصلاحات:} تغییر نهادها
\end{enumerate}
\end{olgoobox}

\begin{noktebox}
\textbf{تعادل:} عدالت باید با ثبات متوازن شود. عفو محدود برای عوامل رده پایین ممکن است ضروری باشد.
\end{noktebox}
