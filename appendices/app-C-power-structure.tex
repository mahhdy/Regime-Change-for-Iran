\section{بیت رهبری}

مرکز اصلی قدرت که شامل:
\begin{itemize}
    \item دفتر رهبری
    \item مشاوران ارشد
    \item نمایندگان ولی فقیه در نهادها
    \item شبکه‌های مالی (ستاد اجرایی فرمان امام)
\end{itemize}

\section{نقاط آسیب‌پذیری}

\begin{enghelabbox}
\begin{enumerate}
    \item \textbf{جانشینی رهبری:} فقدان جانشین واضح
    \item \textbf{وابستگی به نفت:} تحریم‌پذیری
    \item \textbf{تنش جناحی:} رقابت‌های درونی
    \item \textbf{فساد سیستماتیک:} بی‌اعتمادی عمومی
    \item \textbf{انزوای بین‌المللی:} کاهش متحدان
\end{enumerate}
\end{enghelabbox}
